#include <cstdio>

using namespace std;

int main()
{
    int n, m, cnt = 1;
    scanf("%d %d", &n, &m);

    if(n  > 100){
        printf("INPUT ERROR!");
        return 0;
    }
        

    if(m == 1){
        for(int i = 0; i < n; i++){
            for(int j = 0; j < cnt; j++){
                printf("*");
            }
            cnt++;
            printf("\n");
        }
    }
    else if(m == 2){
        cnt = n;
        for(int i = 0; i < n; i++){
            for(int j = 0; j < cnt; j++){
                printf("*");
            }
            cnt--;
            printf("\n");
        }
    }
    
    else if(m == 3){
        int total = (n * 2 - 1);
        int blank = total / 2;
        for(int i = 0; i < n; i++){
            for(int j = 0; j < total; j++){
                if(j >= blank && j < (total - blank))
                    printf("*");
                else if(j >= (total - blank))
                    break;
                else
                    printf(" ");
            }

            blank--;
            printf("\n");
        }
    }
    else{
        printf("INPUT ERROR!");
    }
    
    return 0;

}

void print(int c, char ch = '*') {
    while (c--) printf("%c", ch);

}

void pro1(){
    for (int i = 1; i <= n; i++) print(i);
}

void pro2(){
    for (int i = n; i >= 1; i--) print(i);
}

void pro3(){
    // ----*     4 1
    // ---***    3 3
    // --*****   2 5
    // -*******  1 7
    // ********* 0 9

    // 1. 변수 사용
    int starC = 1, emptyC = n - 1;
    for (int i = 1; i <= n; i++) {
        print(emptyC, ' ');
        emptyC--;

        print(starC);
        starC += 2;
    }

    // 2. one base 규칙
    for (int i = 1; i <= n; i++) {
        print(n - i, ' ');
        print(i * 2 - 1);
    }

    // 3. zero base 규칙
    for (int i = 0; i < n; i++) {
        print(n - i - 1, ' ');
        print(i * 2 + 1);
    }
}

#include <cstdio>
#include <iostream>
#include <vector>
#include <algorithm>

#define MAX_ARR_LEN 20

using namespace std;

int arr[MAX_ARR_LEN][MAX_ARR_LEN] = { 0 };
// int dy[8] = { -1, -1, -1, 0, 1, 1, 1, 0 };
// int dx[8] = { -1, 0, 1, 1, 1, 0, -1, -1 };

// 왼쪽 우선 그 다음 아래 우선이므로 방향은
// 우상, 우, 우하, 하  경로만 탐색해야한다.
int dy[4] = { -1, 0, 1, 1};
int dx[4] = {1, 1, 1, 0};
int main()
{
    int answer_cnt = 0, winner = 0, pos_y, pos_x;
    for(int i = 1; i <= 19; i++){
        for(int j = 1; j <= 19; j++){
            scanf("%2d", &arr[i][j]);
        }
    }

    for(int i = 1; i <= 19; i++){
        for(int j = 1; j <= 19; j++){
            if(arr[i][j] == 1 || arr[i][j] == 2){
                int temp = arr[i][j];
                for(int dir = 0; dir < 4; dir++){
                    answer_cnt = 1;
                    int ny = i + dy[dir];
                    int nx = j + dx[dir];
                    // printf("=======start============\n");
                    // printf("%d %d\n", i, j);
                    // 다음 루트가 범위를 벗어날 경우 다시 해볼 필요 없음
                    if(!(ny >= 1 && ny <= 19 && nx >=1 && nx <=19))
                        continue;
                    // 두번째도 같은 숫자인지 체크
                    if(arr[ny][nx] != temp)
                        continue;
                    // 이미 지나온길 하나만 체크
                    // 육목인데 부분적으로 판단해서 오목으로 볼 수 있으므로 
                    int oy = i - dy[dir];
                    int ox = j - dx[dir];
                    if((oy >= 1 && oy <= 19 && ox >=1 && ox <=19 ))
                        if(arr[oy][ox] == temp)
                        continue;
                    // do while 문이므로 결국 세번째부터 오목인지 판단하는 코드
                    do {
                        
                        if(!(ny >= 1 && ny <= 19 && nx >=1 && nx <=19 ))
                            continue;
                        answer_cnt++;
                        // printf("%d %d current_cnt : %d target : %d\n", ny, nx, answer_cnt, temp);
                        ny = i + (answer_cnt * dy[dir]);
                        nx = j + (answer_cnt * dx[dir]);
                         
                    }while(arr[ny][nx] == temp);
                    
                    if(answer_cnt == 5){
                        winner = temp;
                        pos_y = i;
                        pos_x = j;
                        break;
                    }
                }
            }
            if(winner != 0)
                break; 
        }
        if(winner != 0)
            break;

    }


    printf("%d\n", winner);
    if(winner != 0)
        printf("%d %d\n", pos_y, pos_x);




    return 0;
}

#include <cstdio>
#include <iostream>

#define ARR_MAX_LEN 7
using namespace std;

int arr[ARR_MAX_LEN][ARR_MAX_LEN] = { 0 };
bool isvisited[ARR_MAX_LEN][ARR_MAX_LEN] = { false };

// int dy[8] = {-1, -1, -1, 0, 1, 1, 1, 0};
// int dx[8] = {-1, 0, 1, 1, 1, 0, -1, -1};

// 좌 우선, 하 우선으로 단방향 탐색하여 중복을 없앰
int dy[4] = {-1, 0, 1, 1};
int dx[4] = {1, 1, 1, 0};
int main()
{
    int bingo_cnt = 0, cnt = 0;
    for(int i = 1; i <= 5; i++){
        for(int j = 1; j <= 5; j++){
            scanf("%2d", &arr[i][j]);
        }
    }


    for(int i = 1; i <= 5; i++){
        for(int j = 1; j <= 5; j++){
            int num;
            cnt++;
            scanf("%2d", &num);
            // 빙고판 지우기
            for(int r = 1; r <= 5; r++){
                for(int c = 1; c <= 5; c++){
                    if(arr[r][c] == num)
                        isvisited[r][c] = true;
                }
            }
            // 빙고가 있는지 완전 탐색
            for(int r = 1; r <= 5; r++){
                for(int c = 1; c <= 5; c++){
                    int tmp_bingo_cnt = 0;
                    if(isvisited[r][c]){
                        for(int dir = 0; dir < 4; dir++){
                            int temp_cnt = 1;
                            int ny = r + dy[dir];
                            int nx = c + dx[dir];
                            while((ny >= 1 && ny <= 5 && nx >= 1 && nx <= 5) && isvisited[ny][nx]){
                                temp_cnt++;
                                if(temp_cnt == 5){
                                    tmp_bingo_cnt++;
                                    break;
                                }
                                ny = r + (temp_cnt * dy[dir]);
                                nx = c + (temp_cnt * dx[dir]);
                                    
                            }
                        }
                        
                    }

                    if(tmp_bingo_cnt != 0){
                        bingo_cnt += tmp_bingo_cnt;
                    }
                }
            }
            //bingo count 하기
            // 동시에 bingo_cnt 가 올라가서 3이 넘을 수 있으므로 >= 로 한다
            if(bingo_cnt >= 3){
                printf("%d\n", cnt);
                return 0; 
            }
            else
                bingo_cnt = 0;
        }
    }

    return 0;
}

int T, N, K;
int arr[MAX_NUM] = { 0 };
bool isvisited[MAX_NUM] = { false };
void comb(int cnt)
{
    if(cnt == N){
        for(int i = 0; i < cnt - 1; i++){
            if(arr[i] > arr[i + 1])
                return;
        }
        for(int i = 0; i < cnt; i++){
            printf("%d ", arr[i]);
        }
        printf("\n");
        return ;

    for(int i = 1; i <= K; i++){

}

#ifndef _CRT_SECURE_NO_WARNINGS
#define _CRT_SECURE_NO_WARNINGS
#endif

#include <stdio.h>

int T, N, K;
int A[10];

/*
* 중복 조합
* A[c]를 선택해
*/
void recur1(int c) {
    // base case
    if (c > N) {
        for (int i = 1; i <= N; i++)
            printf("%d ", A[i]);
        printf("\n");
        return;
    }

    // normal case
    // 조건을 만족하는 것 부터, 즉 앞 숫자와 같은 숫자가 다음에 올 수 있다.
    for (int i = A[c-1]; i <= K; i++) {
        A[c] = i;
        recur1(c + 1);
    }
}

/* 조합 */
void recur2(int c) {
    if (c > N) {
        for (int i = 1; i <= N; i++)
            printf("%d ", A[i]);
        printf("\n");

        printf("\n");
        return ;

        arr[cnt] = i;
        perm(cnt + 1);
         
    }
}

{

        return;
    }
    for (int i = 1; i <= 6; i++) {
        if (isvisited[i])

        arr[cnt] = i;
        isvisited[i] = true;
        dup_comb(cnt + 1);

    }

}

        int cy = q.front().first;
        int cx = q.front().second;
        q.pop();

        for (int dir = 0; dir < 4; dir++) {
            int ny = cy + dy[dir];
            int nx = cx + dx[dir];

            if (ny >= 0 && ny < N && nx >= 0 && nx < M) {
                if (arr[ny][nx] == 0 && !isvisited[ny][nx]) {

                    q.push(make_pair(ny, nx));
                }
            }
        }
    }
}
