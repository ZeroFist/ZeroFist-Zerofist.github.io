게리맨더링 2

문제에서 주어진 조건에 맞춰 그대로 구현을 진행한다. 기본적으로 1,2,3,4,5 선거구로 나누고 그 중 기본이 되는 5번 선거구의 경계선과 기준점을 고르는 문제이다.

따라서 기준점 및 경계선을 문제에서 주어진 범위 내에서 자유롭게 선택해서 선거구를 나누고 그 중 인구 차이의 최솟값을 구한다.

#include <iostream>
#include <cstdio>
#include <cmath>
#include <vector>
#include <algorithm>
using namespace std;

int N, answer = 98765432;
int arr[21][21] = { {0, }, };

int sum = 0;
void input_func()
{
	scanf("%d", &N);
	for (int i = 1; i <= N; i++)
	{
		for (int j = 1; j <= N; j++)
		{
			scanf("%d", &arr[i][j]);
			sum += arr[i][j];
		}
	}
}

int diff(int x, int y, int d1, int d2)
{
	vector<int> population(5, 0);
	int temp_arr[21][21] = { {0,}, };
	temp_arr[x][y] = 5;
	//(x, y), (x + 1, y - 1), ..., (x + d1, y - d1)
	//(x + d2, y + d2), (x + d2 + 1, y + d2 - 1), ..., (x + d2 + d1, y + d2 - d1)
	for (int i = 1; i <= d1; i++)
	{
		temp_arr[x + i][y - i] = 5;
		temp_arr[x + d2 + i][y + d2 - i] = 5;
	}
	//(x, y), (x + 1, y + 1), ..., (x + d2, y + d2)
	//(x + d1, y - d1), (x + d1 + 1, y - d1 + 1), ... (x + d1 + d2, y - d1 + d2)
	for(int i = 1; i <= d2; i++)
	{
		temp_arr[x + i][y + i] = 5;
		temp_arr[x + d1 + i][y - d1 + i] = 5;
	}

	//1번 선거구 : 1 ≤ r < x + d1, 1 ≤ c ≤ y
	
	for (int r = 1; r < x + d1; r++)
	{
		for (int c = 1; c <= y; c++)
		{
			if (temp_arr[r][c] == 5)
			{
				break;
			}
			else
			{
				population[0] += arr[r][c];
			}
		}
	}
	//2번 선거구 : 1 ≤ r ≤ x + d2, y < c ≤ N
	for (int r = 1; r <= x + d2; r++)
	{
		for (int c = N; c > y; c--)
		{
			if (temp_arr[r][c] == 5)
			{
				break;
			}
			else
			{
				population[1] += arr[r][c];
			}
		}
	}
	//3번 선거구 : x + d1 ≤ r ≤ N, 1 ≤ c < y - d1 + d2
	for (int r = x + d1; r <= N; r++)
	{
		for (int c = 1 ; c < y - d1 + d2; c++)
		{
			if (temp_arr[r][c] == 5)
			{
				break;
			}
			else
			{
				population[2] += arr[r][c];
			}
		}
	}
	//4번 선거구 : x + d2 < r ≤ N, y - d1 + d2 ≤ c ≤ N
	for (int r = x + d2 + 1; r <= N; r++)
	{
		for (int c = N; c >= y - d1 + d2; c--)
		{
			if (temp_arr[r][c] == 5)
			{
				break;
			}
			else
			{
				population[3] += arr[r][c];
			}
		}
	}
	//for (int r = 1; r <= N; r++)
	//{
	//	for (int c = 1; c <= N; c++)
	//	{
	//		if (temp_arr[r][c] == 5)
	//		{
	//			population[4] += arr[r][c];
	//		}
	//	}
	//}
	population[4] = sum - population[3] - population[2] - population[1] - population[0];
	sort(population.begin(), population.end());

	return population[4] - population[0];
}

void sector()
{
	//(d1, d2 ≥ 1 => 무조건 참, 1 ≤ x( => 무조건 참) < x+d1+d2 ≤ N, 1 ≤ y-d1 < y < y+d2 ≤ N)
	for (int x = 1; x <= N; x++)
	{
		for (int y = 1; y <= N; y++)
		{
			for (int d1 = 1; d1 <= N; d1++)
			{
				for (int d2 = 0; d2 <= N ;d2++)
				{
					if (x + d1 + d2 > N || 1 > y - d1 || y + d2 > N)
						continue;
					answer = min(answer, diff(x, y, d1, d2));

				}
			}
		}
	}
}
/*
* 선거구 나누기
기준점 (x, y)와 경계의 길이 d1, d2를 정한다. (d1, d2 ≥ 1, 1 ≤ x < x+d1+d2 ≤ N, 1 ≤ y-d1 < y < y+d2 ≤ N)
다음 칸은 경계선이다.
(x, y), (x+1, y-1), ..., (x+d1, y-d1)
(x, y), (x+1, y+1), ..., (x+d2, y+d2)
(x+d1, y-d1), (x+d1+1, y-d1+1), ... (x+d1+d2, y-d1+d2)
(x+d2, y+d2), (x+d2+1, y+d2-1), ..., (x+d2+d1, y+d2-d1)
경계선과 경계선의 안에 포함되어있는 곳은 5번 선거구이다.
5번 선거구에 포함되지 않은 구역 (r, c)의 선거구 번호는 다음 기준을 따른다.
1번 선거구: 1 ≤ r < x+d1, 1 ≤ c ≤ y
2번 선거구: 1 ≤ r ≤ x+d2, y < c ≤ N
3번 선거구: x+d1 ≤ r ≤ N, 1 ≤ c < y-d1+d2
4번 선거구: x+d2 < r ≤ N, y-d1+d2 ≤ c ≤ N
*/

int main(int argc, char** argv)
{
	input_func();
	sector();
	printf("%d\n", answer);
}
