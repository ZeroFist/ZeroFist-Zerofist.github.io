연구소

0이 빈칸, 1이 벽, 2는 바이러스가 있는 곳을 나타내며 바이러스는 모든 빈 칸으로 퍼져나갈 수 있다. 이 때 벽을 총 3개 세울 수 있고 바이러스가 퍼질 수 없는 곳이 안전 영역 일 때 안전 영역의 크기의 최댓값을 구하는 문제

바이러스가 상하좌우로 퍼진다 => 넓이 우선 탐색인 bfs을 활용한다. 벽을 총 3개 세울 수 있다. => 조합으로 모든 빈 칸 중 3개를 선택하여 벽을 세우고 그 안전 영역을 구한다.

=> 이 때 임의의 벽을 세우는 새로운 지도를 생성하기 위해 임시 배열로 복사한다.

#include <iostream>
#include <cstdio>
#include <cstring>
#include <queue>
using namespace std;
#define ARR_MAX 8
int N, M;
int arr[ARR_MAX][ARR_MAX] = {0};
int arr_temp[ARR_MAX][ARR_MAX] = {0};
int dx[4] = {0, 0, -1, 1};
int dy[4] = {-1, 1, 0, 0};
int answer = 0;
void bfs()
{
    int spr_result[ARR_MAX][ARR_MAX];

    memcpy(&(spr_result[0][0]), &(arr_temp[0][0]), sizeof(int) * ARR_MAX * ARR_MAX);
    queue<pair<int, int>> q;
    for(int i = 0; i < N; i++)
    {
        for(int j = 0; j < M; j++)
        {
            if(spr_result[i][j] == 2)
            {
                q.push(make_pair(i, j));
            }
        }
    }

    while(!q.empty())
    {
        int x = q.front().first;
        int y = q.front().second;
        q.pop();
        for(int i = 0; i < 4; i++)
        {
            int cx = x + dx[i];
            int cy = y + dy[i];
            if(cx >= 0 && cx < N && cy >= 0 && cy < M)
            {
                if(spr_result[cx][cy] == 0)
                {
                    spr_result[cx][cy] = 2;
                    q.push(make_pair(cx, cy));
                }
            }
        }
    }
    int cnt = 0;
    for(int i = 0; i < N; i++)
    {
        for(int j = 0; j < M; j++)
        {
            if(spr_result[i][j] == 0)
            {
                cnt++;
            }
        }
    }
    answer = max(answer, cnt);
}
//벽을 선택하는 함수
void virus_spread(int cnt)
{

    if(cnt == 3)
    {
        bfs();
        return;
    }

    for(int i = 0; i < N; i++)
    {
        for(int j = 0; j < M; j++)
        {
            if(arr_temp[i][j] == 0)
            {
                arr_temp[i][j] = 1;
                virus_spread(cnt+1);
                arr_temp[i][j] = 0;
            }
        }
    }
}
int main()
{
    scanf("%d %d", &N, &M);

    for(int i = 0; i < N; i++)
    {
        for(int j = 0; j < M; j++)
        {
            scanf("%d", &arr[i][j]);
        }
    }

    for(int i = 0; i < N; i++)
    {
        for(int j = 0; j < M; j++)
        {
            //벽을 새우는 모든 경우의 수를 따져본다
            if(arr[i][j] == 0)
            {
                memcpy(&(arr_temp[0][0]), &(arr[0][0]), sizeof(int) * ARR_MAX * ARR_MAX);
                arr_temp[i][j] = 1; // 벽 한개를 세운다.
                virus_spread(1);
                arr_temp[i][j] = 0;
            }


        }
    }
    cout << answer << endl;

   return 0;
}
