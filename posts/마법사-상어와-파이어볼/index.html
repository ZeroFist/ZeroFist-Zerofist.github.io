마법사 상어와 파이어볼

모든 파이어볼이 자신의 방향 di로 속력 si칸 만큼 이동한다. 이동하는 중에는 같은 칸에 여러 개의 파이어볼이 있을 수도 있다. 이동이 모두 끝난 뒤, 2개 이상의 파이어볼이 있는 칸에서는 다음과 같은 일이 일어난다. 같은 칸에 있는 파이어볼은 모두 하나로 합쳐진다. 파이어볼은 4개의 파이어볼로 나누어진다. 나누어진 파이어볼의 질량, 속력, 방향은 다음과 같다. 질량은 ⌊(합쳐진 파이어볼 질량의 합)/5⌋이다. 속력은 ⌊(합쳐진 파이어볼 속력의 합)/(합쳐진 파이어볼의 개수)⌋이다. 합쳐지는 파이어볼의 방향이 모두 홀수이거나 모두 짝수이면, 방향은 0, 2, 4, 6이 되고, 그렇지 않으면 1, 3, 5, 7이 된다. 질량이 0인 파이어볼은 소멸되어 없어진다.

마법사 상어가 이동을 K번 명령한 후, 남아있는 파이어볼 질량의 합을 구해보자.

#include <iostream>
#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>


#define LEN_2D_MAX 50

	using namespace std;

int dy[8] = { -1, -1, 0, 1, 1, 1, 0, -1 };
int dx[8] = { 0, 1, 1, 1, 0, -1, -1, -1 };

int N, M, K, answer = 0;



struct fireball_info
{
	int y;
	int x;
	int m;
	int s;
	int d;
};

vector<fireball_info> fb;
vector<int> arr[LEN_2D_MAX][LEN_2D_MAX];


void fmove()
{
	vector<int> temp_arr[LEN_2D_MAX][LEN_2D_MAX];
	for (int i = 0; i < fb.size(); i++)
	{
		int dir = fb[i].d;
		// 속력이 배열의 크기를 넘어갈 경우를 생각한다.
		int speed = fb[i].s % N;
		// 이동방향에 따라 좌표값이 음수가 될 수 있다, 이를 방지하기 위해 N을 더해준다.
		int ny = (fb[i].y + (dy[dir] * speed) + N) % N;
		int nx = (fb[i].x + (dx[dir] * speed) + N) % N;
		temp_arr[ny][nx].push_back(i);
		// 해당 파이어볼의 이동 구현
		fb[i].y = ny;
		fb[i].x = nx;
	}

	// 원본에 내용 적용

	for (int i = 0; i < N; i++)
	{
		for (int j = 0; j < N; j++)
		{
			arr[i][j] = temp_arr[i][j];
		}
	}
	return;
}

// 합쳐지는 파이어 볼의 질량을 계산하는 곳

void fsum()
{
	vector<fireball_info> temp_fb;
	for (int i = 0; i < N; i++)
	{
		for (int j = 0; j < N; j++)
		{
			//해당 위치에 파이어볼이 없을때
			if (arr[i][j].size() == 0)
				continue;
			// 해당위치에 파이어볼이 하나만 있을때 충돌 x
			// 그 위치에 파이어볼을 넣어주면된다
			if (arr[i][j].size() == 1)
			{
				int idx = arr[i][j][0];
				temp_fb.push_back(fb[idx]);
				continue;
			}
			int msum = 0, ssum = 0;
			bool is_odd = true, is_even = true;
			// 파이어볼이 중첩되는 경우
			for (int k = 0; k < arr[i][j].size(); k++)
			{
				int idx = arr[i][j][k];
				msum += fb[idx].m;
				ssum += fb[idx].s;
				if ((fb[idx].d & 1) == 0)
				{
					is_odd = false;
				}
				else
				{
					is_even = false;
				}
			}
			// 질량이 0인 파이어볼 소멸
			if (msum / 5 == 0)
				continue;
			int temp_m = msum / 5;
			int temp_s = ssum / arr[i][j].size();
			// 합쳐진 파이업볼은 4개의 파이어볼로 4방향으로 퍼져나간다.
			for (int k = 0; k < 4; k++)
			{
				// 모두 홀수이거나 모두 짝수일 때
				if (is_odd || is_even)
				{
					temp_fb.push_back({ i, j, temp_m, temp_s, k * 2 });
				}
				else
				{
					temp_fb.push_back({ i, j, temp_m, temp_s, ((k * 2) + 1) });
				}
			}
		}
	}
	// 벡터의 경우 대입연산이 가능
	// 대입연산 수행시 대입 받는 좌변 벡터는 우변 벡터의 크기만큼 자동으로 늘어나고 우변의 모든 요소가 좌변으로 대입된다.
	// 원래 좌변에 들어가있는 값은 파괴되고 우변 벡터와 똑같은 값, 크기를 가지게 된다.
	fb = temp_fb;
	return;
}

void simulate()
{
	while (K--)
	{
		fmove();
		fsum();
	}


	for (int i = 0; i < fb.size(); i++)
	{
		answer += fb[i].m;
	}

	return;
}

int main(int argc, char** argv)
{
	scanf("%d %d %d", &N, &M, &K);
	for (int i = 0; i < M; i++)
	{
		int y, x, m, s, d;
		scanf("%d %d %d %d %d", &y, &x, &m, &s, &d);
		y--;
		x--;
		fb.push_back({ y, x, m, s, d });
		// 2차원 좌표에서 파이어볼 의 index를 저장한다
		arr[y][x].push_back(i);


	}

	simulate();

	printf("%d\n", answer);
	return 0;
}
