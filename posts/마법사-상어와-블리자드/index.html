마법사 상어와 블리자드

  1. 구슬이 블리자드 마법에 맞아 파괴되는 내용 구현
  2. 연속하는 구슬이 있을 때 폭발하는 내용 구현
  3. 구슬이 변하는 것 구현(A 구슬의 개수, B 구슬의 번호)

핵심은 2차원 배열 내에서 나선형으로 진행하기 때문에 2차원 => 1차원 배열로 해당 내용을 구현하고 그 내용을 다시 1차원 => 2차원 배열로 복원시켜 해당 동작을 구현하는 것

#include <iostream>
#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>

#define LEN_1D_MAX 2500
#define LEN_2D_MAX 49
using namespace std;

int N, M, answer = 0;

int line_size = 0;
int line[LEN_1D_MAX] = { 0, };
int arr[LEN_2D_MAX][LEN_2D_MAX] = { 0, };

int dy[4] = { -1, 1, 0, 0 };
int dx[4] = { 0, 0, -1, 1 };
void destroy_by_rule(int d, int s)
{
	int cy = N / 2;
	int cx = N / 2;

	for (int i = 1; i <= s; i++)
	{
		int ny = cy + (dy[d] * i);
		int nx = cx + (dx[d] * i);
		arr[ny][nx] = 0;
	}
}

void convert_to_line()
{
	int cy = N / 2;
	int cx = N / 2;
	
	int ddy[4] = { 0, 1, 0, -1 };
	int ddx[4] = { -1, 0, 1, 0 };

	int dir = 0, loop_cnt = 2;

	line_size = 0;

	for (int move_r = 1; move_r < N; move_r++)
	{
		// 마지막의 경우에만 loop cnt 가 3번으로 늘어난다
		if (move_r == N - 1)
		{
			loop_cnt = 3;
		}
		for (int i = 0; i < loop_cnt; i++)
		{
			for (int j = 0; j < move_r; j++)
			{
				cy += ddy[dir];
				cx += ddx[dir];
				if (arr[cy][cx] != 0)
				{
					line[line_size++] = arr[cy][cx];
				}
			}
			dir = (dir + 1) % 4;
		}

		

	}
	return;
}

int explosion()
{
	int point = 0;
	int cnt = 1;

	for (int i = 1; i < line_size; i++)
	{
		if (line[i - 1] == line[i])
		{
			++cnt;
		}
		else
		{
			if (cnt >= 4)
			{
				for (int j = 1; j <= cnt; j++)
				{
					point += line[i - j];
					line[i - j] = 0;
				}
			}
			cnt = 1;
		}
	}
	// 마지막까지 동일한 숫자가 반복되고 4가 넘는 경우에 대한 예외처리를 해준다
	if (cnt >= 4)
	{
		for (int j = 1; j <= cnt; j++)
		{
			point += line[line_size - j];
			line[line_size - j] = 0;
		}
	}

	answer += point;
	return point;
}

void remove_zero()
{
	int back_size = 0;
	int backup[2500] = { 0, };

	for (int i = 0; i < line_size; i++)
	{
		if (line[i] != 0)
		{
			backup[back_size++] = line[i];
		}
	}
	for (int i = 0; i < back_size; i++)
	{
		line[i] = backup[i];
	}
	line_size = back_size;

	return;
}

void change_by_rule()
{
	int back_size = 0;
	int backup[2500] = { 0, };
	int cnt = 1;
	for (int i = 1; i < line_size; i++)
	{
		if (line[i - 1] == line[i])
		{
			++cnt;
		}
		else
		{
			if (back_size < N * N - 2)
			{
				// 구슬의 개수와 최초 그 구슬의 번호를 번갈아가며 출력
				backup[back_size++] = cnt;
				backup[back_size++] = line[i - 1];
			}
			cnt = 1;
		}
	}
	if (line_size > 0 && back_size < N * N - 2)
	{
		backup[back_size++] = cnt;
		backup[back_size++] = line[line_size - 1];
	}
	for (int i = 0; i < back_size; i++)
	{
		line[i] = backup[i];
	}
	line_size = back_size;
	return;
}

void convert_to_mat()
{
	int cy = N / 2;
	int cx = N / 2;

	int ddy[4] = { 0, 1, 0, -1 };
	int ddx[4] = { -1, 0, 1, 0 };

	int dir = 0, loop_cnt = 2;
	int cur = 0;


	for (int move_r = 1; move_r < N; move_r++)
	{
		// 마지막의 경우에만 loop cnt 가 3번으로 늘어난다
		if (move_r == N - 1)
		{
			loop_cnt = 3;
		}
		for (int i = 0; i < loop_cnt; i++)
		{
			for (int j = 0; j < move_r; j++)
			{
				cy += ddy[dir];
				cx += ddx[dir];
				if (cur < line_size)
				{
					arr[cy][cx] = line[cur++];
				}
				else
				{
					arr[cy][cx] = 0;
				}
			}
			dir = (dir + 1) % 4;
		}

		

	}

	return;
}


int main(int argc, char** argv)
{
	scanf("%d %d", &N, &M);

	for (int i = 0; i < N; i++)
	{
		for (int j = 0; j < N; j++)
		{
			scanf("%d", &arr[i][j]);
		}
	}
	answer = 0;
	for (int i = 0; i < M; i++)
	{
		int d, s;
		scanf("%d %d", &d, &s);
		destroy_by_rule(d - 1, s);
		convert_to_line();
		// 구슬이 한번 파괴되고 나면 배열이 재정비되고 그 이후에도 또다시 파괴시켜야할 구슬이 있을 수 있기 때문에 파괴될 구슬이 없을 때까지 반복해서 수행한다.
		while (explosion() != 0)
		{
			remove_zero();
		}
		remove_zero();
		change_by_rule();
		convert_to_mat();
	}



	printf("%d\n", answer);
	return 0;
}
